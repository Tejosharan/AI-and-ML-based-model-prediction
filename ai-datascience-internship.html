# inference_pipeline.py — Live prediction pipeline
import joblib
from fastapi import FastAPI
from pydantic import BaseModel
import pandas as pd

app = FastAPI(title="ML Inference API")

# Load persisted models and preprocessors
health_model  = joblib.load("models/health_tf_pipeline.pkl")
house_model   = joblib.load("models/house_stack_pipeline.pkl")
health_scaler = joblib.load("models/health_scaler.pkl")
house_scaler  = joblib.load("models/house_scaler.pkl")

class HealthInput(BaseModel):
    age: float; bmi: float; glucose: float
    systolic_bp: float; cholesterol: float

class HouseInput(BaseModel):
    total_sqft: float; overall_qual: int
    neighborhood: str; year_built: int; garage_area: float

@app.post("/predict/health")
def predict_health(data: HealthInput):
    df = pd.DataFrame([data.dict()])
    df = engineer_features(df)
    X  = health_scaler.transform(df)
    prob = health_model.predict(X)[0][0]
    return { "risk_probability": float(prob),
             "outcome": "high" if prob > 0.5 else "low" }

@app.post("/predict/house")
def predict_house(data: HouseInput):
    df = pd.DataFrame([data.dict()])
    df = engineer_housing_features(df)
    X  = house_scaler.transform(df)
    price = house_model.predict(X)[0]
    return { "estimated_price": round(float(price), 2),
             "currency": "INR" }
